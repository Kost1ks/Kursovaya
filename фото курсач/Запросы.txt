// Выбрать общую сумму заказов по каждому клиенту
SELECT 
    Клиенты.Имя, 
    Клиенты.Фамилия, 
    SUM(Заказы.Общая_Сумма) AS Общая_Сумма_Заказов
FROM 
    Заказы
JOIN 
    Клиенты ON Заказы.ID_Клиента = Клиенты.ID_Клиента
GROUP BY 
    Клиенты.Имя, Клиенты.Фамилия;
// Удалить все товары, которые больше не продаются
DELETE FROM Товары
WHERE На_Складе = 0;
// Выбрать все детали заказа по идентификатору заказа
SELECT * FROM Детали_Заказа WHERE ID_Заказа = 1;
// Выбрать все товары в ценновой категории от 10 до 20 рублей
SELECT * FROM Товары WHERE Цена BETWEEN 10 AND 20;
// Вывод информации о пользователи по имени Костя
SELECT 
*FROM Клиенты
WHERE Имя = 'Костя';
// Вывод информации имена клиентов по ID
SELECT ID_Клиента, Имя
FROM Клиенты; 
// Вывод информации о заказе определенной даты
SELECT *
FROM Заказы
WHERE YEAR(Дата_Заказа) = 2024 AND MONTH(Дата_Заказа) = 05; 
// Вывод информации о заказе сумма которого находиться в диапозоне от 100 до 200 рублей
SELECT *
FROM Заказы
WHERE Общая_Сумма BETWEEN 100 AND 200; 
// Выбрать максимальную и минимальную цену товаров
SELECT MAX(Цена) AS Максимальная_Цена, MIN(Цена) AS Минимальная_Цена FROM Товары;
// Выбрать среднюю цену всех товаров
SELECT AVG(Цена) AS Средняя_Цена FROM Товары;
